340  bool isToleranced() const
341  {
342  // Check if they are empty
343  if (lower_tolerance.size() == 0 || upper_tolerance.size() == 0)
344  return false;
345 
346  // Check if they are the same
347  static auto max_diff = static_cast<double>(std::numeric_limits<float>::epsilon());
348 
349  if ((lower_tolerance.array() > max_diff).any())
350  throw std::runtime_error("CartesianWaypoint: lower tolerance was provided but must be <= 0,");
351 
352  if ((upper_tolerance.array() < -max_diff).any())
353  throw std::runtime_error("CartesianWaypoint: upper tolerance was provided but must be >= 0,");
354 
355  return !tesseract_common::almostEqualRelativeAndAbs(lower_tolerance, upper_tolerance, max_diff);
356  }
