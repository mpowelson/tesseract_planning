352  bool isToleranced() const
353  {
354  // Check if they are empty
355  if (lower_tolerance.size() == 0 || upper_tolerance.size() == 0)
356  return false;
357 
358  // Check if they are the same

361  if ((lower_tolerance.array() > max_diff).any())
362  throw std::runtime_error("JointWaypoint: lower tolerance was provided but must be <= 0,");
363 
364  if ((upper_tolerance.array() < -max_diff).any())
365  throw std::runtime_error("JointWaypoint: upper tolerance was provided but must be >= 0,");
366 
367  return !tesseract_common::almostEqualRelativeAndAbs(lower_tolerance, upper_tolerance, max_diff);
368  }
