165 template <typename FloatType>
166 void DescartesDefaultPlanProfile<FloatType>::apply(DescartesProblem<FloatType>& prob,
167  const Eigen::Isometry3d& cartesian_waypoint,
168  const Instruction& parent_instruction,
169  const ManipulatorInfo& manip_info,
170  const std::vector<std::string>& active_links,
171  int index) const
172 {
173  assert(isPlanInstruction(parent_instruction));
174  const auto& base_instruction = parent_instruction.as<PlanInstruction>();
175  assert(!(manip_info.empty() && base_instruction.getManipulatorInfo().empty()));
176  ManipulatorInfo mi = manip_info.getCombined(base_instruction.getManipulatorInfo());
177  Eigen::Isometry3d tcp = prob.env->findTCP(mi);
178 
179  /* Check if this cartesian waypoint is dynamic
180  * (i.e. defined relative to a frame that will move with the kinematic chain) */
181  auto it = std::find(active_links.begin(), active_links.end(), prob.manip_inv_kin->getBaseLinkName());
182  if (it != active_links.end() && prob.manip_inv_kin->getBaseLinkName() != mi.working_frame)
183  throw std::runtime_error("DescartesDefaultPlanProfile: Assigned dynamic waypoint but parent instruction working is "
184  "not set to the base link of manipulator!");
185 
186  DescartesCollision::Ptr ci = nullptr;
187  if (enable_collision)
188  ci = std::make_shared<DescartesCollision>(
189  prob.env, active_links, prob.manip_inv_kin->getJointNames(), vertex_collision_check_config, debug);
190 
191  Eigen::Isometry3d manip_baselink_to_waypoint = Eigen::Isometry3d::Identity();
192  if (it == active_links.end())
193  {
194  // Check if the waypoint is not relative to the manipulator base coordinate system
195  Eigen::Isometry3d world_to_waypoint = cartesian_waypoint;
196  if (!mi.working_frame.empty())
197  world_to_waypoint = prob.env_state->link_transforms.at(mi.working_frame) * cartesian_waypoint;
198 
199  Eigen::Isometry3d world_to_base_link = prob.env_state->link_transforms.at(prob.manip_inv_kin->getBaseLinkName());
200  manip_baselink_to_waypoint = world_to_base_link.inverse() * world_to_waypoint;
201  }
202 
203  // Add vertex evaluator
204  std::shared_ptr<descartes_light::WaypointSampler<FloatType>> sampler;
205  if (vertex_evaluator == nullptr)
206  {
207  auto ve = std::make_shared<DescartesJointLimitsVertexEvaluator>(prob.manip_inv_kin->getLimits().joint_limits);
208  sampler = std::make_shared<DescartesRobotSampler<FloatType>>(
209  manip_baselink_to_waypoint, target_pose_sampler, prob.manip_inv_kin, ci, tcp, allow_collision, ve);
210  }
211  else
212  {
213  sampler = std::make_shared<DescartesRobotSampler<FloatType>>(manip_baselink_to_waypoint,
214  target_pose_sampler,
215  prob.manip_inv_kin,
216  ci,
217  tcp,
218  allow_collision,
219  vertex_evaluator(prob));
220  }
221  prob.samplers.push_back(std::move(sampler));
222 
223  if (index != 0)
224  {
225  // Add edge Evaluator
226  if (edge_evaluator == nullptr)
227  {
228  if (enable_edge_collision)
229  {
230  auto compound_evaluator = std::make_shared<descartes_light::CompoundEdgeEvaluator<FloatType>>();
231  compound_evaluator->evaluators.push_back(
232  std::make_shared<descartes_light::EuclideanDistanceEdgeEvaluator<FloatType>>());
233  compound_evaluator->evaluators.push_back(
234  std::make_shared<DescartesCollisionEdgeEvaluator<FloatType>>(prob.env,
235  active_links,
236  prob.manip_inv_kin->getJointNames(),
237  edge_collision_check_config,
238  allow_collision,
239  debug));
240  prob.edge_evaluators.push_back(compound_evaluator);
241  }
242  else
243  {
244  prob.edge_evaluators.push_back(std::make_shared<descartes_light::EuclideanDistanceEdgeEvaluator<FloatType>>());
245  }
246  }
247  else
248  {
249  prob.edge_evaluators.push_back(edge_evaluator(prob));
250  }
251  }
252 
253  prob.num_threads = num_threads;
254 }
